Once the database is running, you can go to mysql prompt and create the database and tables

CREATE database customer_inventories;
use customer_inventories;
CREATE TABLE customer_details(
   customerId INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
   customerName VARCHAR(50) COLLATE utf8mb4_general_ci NOT NULL,
   customerAddress VARCHAR(500) COLLATE utf8mb4_general_ci NOT NULL,
   customerContactEmailAddress VARCHAR(50) COLLATE utf8mb4_general_ci NOT NULL,
   customerContactPhoneNumber VARCHAR(50) COLLATE utf8mb4_general_ci NOT NULL,
   customerContractId VARCHAR(50) COLLATE utf8mb4_general_ci NOT NULL,
   customerContractEndDate DATE NOT NULL   
);
CREATE TABLE customer_trucks(
   truckRegistrationNumber VARCHAR(50) COLLATE utf8mb4_general_ci PRIMARY KEY     NOT NULL,
   customerIdTruck           INT    NOT NULL,
   truckInsurancePolicyNumber INT NOT NULL,
   truckMakeModel VARCHAR(50) COLLATE utf8mb4_general_ci NOT NULL,
   INDEX (customerIdTruck)
);
